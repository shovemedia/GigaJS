1 define(function(require){
  2 
  3 	var Q = require('lib/q');
  4 	var signals = require('lib/signals');
  5 	var History = require('lib/History');
  6 
  7 	var SiteController = require('lib/giga/SiteController');
  8 	var FlowController = require('lib/giga/FlowController');
  9 	var TransitionController = require('lib/giga/TransitionController');
 10 
 11 	var Giga = function($context, $hidden)
 12 	{
 13 		var self = this;
 14 
 15 		this.$context = $context;
 16 		this.$hidden = $hidden;
 17 
 18 		this.flowController = new FlowController();
 19 		this.siteController = new SiteController(this.flowController);
 20 
 21 		this.transitionController = new TransitionController(this.$context, this.$hidden);
 22 
 23 		this.currentBranch = '/';
 24 		this.targetBranch = null;
 25 
 26 		History.Adapter.bind(window, 'statechange', function() {
 27 			var State = History.getState(); 
 28 			History.log(State.data, State.title, State.url);
 29 
 30 			self.gotoBranch(History.getShortUrl(State.url));	
 31 		});
 32 	};
 33 
 34 
 35 	var p = Giga.prototype;
 36 
 37 	p.init = function()
 38 	{
 39 		var self = this;
 40 
 41 		this.siteController.init(this.flowController.defaultFlow);
 42 
 43 		this.siteController.on.transitionOut.add(function(step){
 44 			var content = self.contentRenderer.getOutgoingContent(self.transitioningBranch);
 45 			self.transitionController.xout(content, self.transitioningBranch, step);
 46 		});
 47 		this.siteController.on.transitionIn.add(function(step){
 48 			var content = self.contentRenderer.getIngoingContent(self.transitioningBranch);
 49 			self.transitionController.xin(content, self.transitioningBranch, step);
 50 		});
 51 
 52 		this.siteController.on.preload.add(function(step){
 53 			//console.log('preload step hold!');
 54 			step.hold();
 55 
 56 			var promise = self.preloadController.get(self.transitioningBranch, step);
 57 
 58 			promise.then(function(x){
 59 				//console.log('preload step release!');
 60 
 61 				if (x.parent().length == 0)
 62 				{
 63 					self.contentRenderer.addContent(x);
 64 				}	
 65 
 66 				step.release();
 67 			})
 68 		});
 69 
 70 		this.siteController.on.complete.add(function(){
 71 			self.currentBranch = self.transitioningBranch;
 72 			self.navigateTo(self.targetBranch); //
 73 		});
 74 
 75 		this.gotoBranch(History.getShortUrl(History.getLocationHref()));
 76 	};
 77

 97 	// This method is the beginning of the event chain
 98 	p.gotoBranch = function(branch)
 99 	{
100 		if (!branch) {
101 			branch = "/";
102 		}
103 
104 		branch = this.normalizeBranch(branch);
105 		
106 		if (this.targetBranch != branch)
107 		{
108 			console.log('Giga goto', 'branch', branch, 'targetBranch', this.targetBranch);
109 			//console.trace();
110 
111 			this.targetBranch = branch;
112 
113 			if (!this.siteController.isBusy())
114 			{
115 				this.navigateTo(branch);
116 			}
117 		}	
118 	};
119 
120 	p.navigateTo = function(branch)
121 	{
122 		if (this.transitioningBranch != branch)
123 		{
124 			this.transitioningBranch = branch;
125 
126 
127 			console.log('navigate', 'from', this.currentBranch, 'to', this.transitioningBranch);
128 
129 			var currentBranchArray = this.currentBranch.split('/');
130 			var transitioningBranchArray = this.transitioningBranch.split('/');
131 
132 			var rootChangeBranch = '';
133 
134 			for (var i=0, len = transitioningBranchArray.length; i<len; i++)
135 			{
136 				rootChangeBranch += transitioningBranchArray[i] + '/';
137 				if (currentBranchArray[i] != transitioningBranchArray[i])
138 				{
139 					break;
140 				}	
141 			}
142 
143 			console.log('rootChangeBranch', rootChangeBranch);
144 
145 			var pageFlow = this.flowController.getBranchFlow(rootChangeBranch);
146 			if (pageFlow != undefined)
147 			{
148 				this.siteController.init(pageFlow);
149 			}
150 			else
151 			{
152 				this.siteController.init(this.flowController.defaultFlow);
153 			}
154 
155 			this.siteController.processPageChange();
156 		}
157 	};
158 
159 
160 	p.normalizeBranch = function(branch)
161 	{
162 		//force end with '/'
163 		if (branch.indexOf('/', branch.length - 1) == -1)
164 		{
165 			branch += '/';
166 		}	
167 
168 		return branch;
169 	}
