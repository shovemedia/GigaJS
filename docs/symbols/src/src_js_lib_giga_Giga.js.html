214 	p.gotoBranch = function(branch)
215 	{
216 		if (!branch) {
217 			branch = "/";
218 		}
219 
220 		branch = this.normalizeBranch(branch);
221 		
222 		if (this.targetBranch != branch)
223 		{
224 			console.log('Giga goto', 'branch', branch, 'targetBranch', this.targetBranch);
225 			//console.trace();
226 
227 			this.targetBranch = branch;
228 
229 			if (!this.siteController.isBusy())
230 			{
231 				this.navigateTo(branch);
232 			}
233 		}

236 	p.navigateTo = function(branch)
237 	{
238 		if (this.transitioningBranch != branch)
239 		{
240 			this.transitioningBranch = branch;
241 
242 
243 			console.log('navigate', 'from', this.currentBranch, 'to', this.transitioningBranch);
244 
245 			var currentBranchArray = this.currentBranch.split('/');
246 			var transitioningBranchArray = this.transitioningBranch.split('/');
247 
248 			this.rootChangeBranch = '';
249 
250 			for (var i=0, len = transitioningBranchArray.length; i<len; i++)
251 			{
252 				this.rootChangeBranch += transitioningBranchArray[i] + '/';
253 				if (currentBranchArray[i] != transitioningBranchArray[i])
254 				{
255 					break;
256 				}	
257 			}
258 
259 			console.log('rootChangeBranch', this.rootChangeBranch);
260 
261 			var pageFlow = this.flowController.getBranchFlow(this.rootChangeBranch);
262 			if (pageFlow != undefined)
263 			{
264 				this.siteController.init(pageFlow);
265 			}
266 			else
267 			{
268 				this.siteController.init(this.flowController.defaultFlow);
269 			}
270 
271 			this.siteController.processPageChange();
272 		}
273 	};
