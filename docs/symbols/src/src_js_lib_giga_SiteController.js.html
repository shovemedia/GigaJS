1 define(function(require){
  2 	
  3 	var StateMachine = require('lib/giga/StateMachine');
  4 	var Step = require('lib/giga/Step');
  5 
  6 	var signals = require('lib/signals');
  7 
  8 	var SiteController = function(flowController){
  9 		this.flowController = flowController;
 10 
 11 		this.transitionOutStep = new Step(); //'x out'
 12 		this.preloadStep = new Step(); //'preload'
 13 		this.transitionInStep = new Step(); //'x in'
 14 
 15 		var self = this;
 16 
 17 		this.transitionOutStep.next = 
 18 		this.preloadStep.next =  
 19 		this.transitionInStep.next = this.nextStep.bind(this);
 20 
 21 		this.on = {
 22 			transitionOut: new signals.Signal(),
 23 			transitionIn: new signals.Signal(),			
 24 			preload: new signals.Signal(),
 25 			complete: new signals.Signal()		
 26 		};
 27 
 28 		this.currentStep = null;
 29 	};
 30 
 31 	var p = SiteController.prototype;
 32 
 33 	p.init = function(flow)
 34 	{
 35 		if (flow == undefined || this.currentFlow != flow)
 36 		{
 37 			switch (flow)
 38 			{
 39 				case this.flowController.PRELOAD_FLOW:
 40 					this.currentFlow = this.flowController.PRELOAD_FLOW;
 41 					this.initPreloadFlow();
 42 					this.initStateListeners();
 43 					break;
 44 
 45 				case this.flowController.REVERSE_FLOW:
 46 					this.currentFlow = this.flowController.REVERSE_FLOW;
 47 					this.initReverseFlow();
 48 					this.initCrossListeners();
 49 					break;
 50 
 51 				case this.flowController.CROSS_FLOW:
 52 					this.currentFlow = this.flowController.CROSS_FLOW;
 53 					this.initCrossFlow();
 54 					this.initStateCrossListeners();
 55 					break;
 56 
 57 				case this.flowController.NORMAL_FLOW:
 58 				default:
 59 					this.currentFlow = this.flowController.NORMAL_FLOW;
 60 					this.initNormalFlow();
 61 					this.initStateListeners();
 62 			}
 63 		}
 64 	};
 65 
 66 	p.initNormalFlow = function()
 67 	{
 68 		this.stateMachine = new StateMachine('start');
 69 		this.stateMachine.setTransitionsFor(
 70 			'start', {
 71 				'continue': 'transitionOut'
 72 			});
 73 
 74 		this.stateMachine.setTransitionsFor(
 75 			'transitionOut', {
 76 				'continue': 'preload'
 77 			});
 78 
 79 		this.stateMachine.setTransitionsFor(
 80 			'preload', {
 81 				'continue': 'transitionIn'
 82 			});
 83 
 84 		this.stateMachine.setTransitionsFor(
 85 			'transitionIn', {
 86 				'continue': 'complete'
 87 			});			
 88 
 89 		this.stateMachine.setTransitionsFor(
 90 			'complete', {
 91 				'reset': 'start'
 92 			});
 93 	};
 94

114 			'transitionIn', {

117

121 			});

125 	{
126 		this.stateMachine = new StateMachine('start');

131

154 	{
155 		this.stateMachine = new StateMachine('start');
156 		this.stateMachine.setTransitionsFor(

158 				'continue': 'preload'
159 			});
160 
161 		this.stateMachine.setTransitionsFor(
162 			'preload', {
163 				'continue': 'transition'
164 			});
165 
166 		this.stateMachine.setTransitionsFor(
167 			'transition', {
168 				'continue': 'complete'

177 	p.initStateListeners = function()
178 	{
179 		var self = this;
180 
181 		this.stateMachine.on.transitionOut.add(function(){
182 			console.log('SiteController transition out', self);
183 			self.currentStep = self.transitionOutStep;
184 			self.on.transitionOut.dispatch(self.currentStep);
185 			self.currentStep.check();
186 		});
187 
188 		this.stateMachine.on.preload.add(function(){
189 			console.log('SiteController preload');
190 			self.currentStep = self.preloadStep;
191 			self.on.preload.dispatch(self.currentStep);
192 			self.currentStep.check();
193 		});
194 
195 		this.stateMachine.on.transitionIn.add(function(){
196 			console.log('SiteController transition in');
197 			self.currentStep = self.transitionInStep;
198 			self.on.transitionIn.dispatch(self.currentStep);
199 			self.currentStep.check();
200 		});
201 
202 		this.stateMachine.on.complete.add(function(){
203 			console.log('SiteController complete');
204 			self.currentStep = null;
205 			self.stateMachine.trigger('reset');
206 			self.on.complete.dispatch();
207 		});
208 	};
209 
210 	p.initStateCrossListeners = function()
211 	{
212 		var self = this;
213 
214 		this.stateMachine.on.preload.add(function(){
215 			console.log('SiteController preload');
216 			self.currentStep = self.preloadStep;
217 			self.on.preload.dispatch(self.currentStep);
218 			self.currentStep.check();
219 		});
220 
221 		this.stateMachine.on.transition.add(function(){
222 			console.log('SiteController transition');
223 			self.currentStep = self.transitionInStep;
224 			self.on.transitionIn.dispatch(self.currentStep);
225 			self.on.transitionOut.dispatch(self.currentStep);
226 			self.currentStep.check();
227 		});
228 
229 		this.stateMachine.on.complete.add(function(){
230 			console.log('SiteController complete');
231 			self.currentStep = null;
232 			self.stateMachine.trigger('reset');
233 			self.on.complete.dispatch();
234 		});
