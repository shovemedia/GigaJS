 41 /**
 42 @function
 43 @name Site.prototype.init
 44 @param {$context} a DOM context  ..... 
 45 
 46 @description
 47 The sample site instantiates Giga,
 48 registers the ContentRenderer,
 49 sets the PreloadController,
 50 registers the Transitions
 51 and calls Giga.init
 52 Once upon an example link {@link Site}.
 53 
 54 
 55 @example
 56 var mySite = new Site();
 57 mySite.init($('#context'));
 58 */
 59     p.init = function($context){
 60 		var self = this;
 61 
 62 		var $hidden = $(Site.hiddenTemplate); 
 63 		$context.append($hidden);
 64 
 65 		this.giga = new Giga($context, $hidden);
 66 		this.giga.setContentRenderer( ContentRenderer );
 67 		this.giga.setPreloadController( PreloadController );
 68 		this.giga.registerTransitions( TestTransitions );
 69 
 70 		this.giga.setBranchFlow('/site/project2/', this.giga.flowController.CROSS_FLOW);
 71 		this.giga.setBranchFlow('/site/project3/', this.giga.flowController.CROSS_FLOW);		
 72 
 73 		this.giga.init();
 74 	};
